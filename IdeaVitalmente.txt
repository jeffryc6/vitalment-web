        contenido: 'El metabolismo se ve condicionado por nuestros hábitos diarios...',
        autor: 'Dra. Maricel Muñoz'
    }
];

const reservas = [];

// Rutas de la API

// Obtener todos los artículos/historias
app.get('/api/posts', (req, res) => {
    res.json(posts);
});

// Endpoint para reservar una consulta
app.post('/api/reservas', (req, res) => {
    const { nombre, email, fecha, motivo } = req.body;
    
    if (!nombre || !email || !fecha) {
        return res.status(400).json({ error: 'Por favor completa todos los campos requeridos.' });
    }

    const nuevaReserva = { id: reservas.length + 1, nombre, email, fecha, motivo };
    reservas.push(nuevaReserva);
    
    res.status(201).json({ mensaje: 'Consulta reservada con éxito. Te contactaremos pronto.', reserva: nuevaReserva });
});

app.listen(PORT, () => {
    console.log(`Servidor de Transformación Integral corriendo en el puerto ${PORT}`);
});
